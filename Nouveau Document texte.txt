Créer la table source_orders

CREATE TABLE source_orders (after ROW<order_id STRING, customer_id STRING, order_status STRING>) WITH ('connector' = 'kafka', 'topic' = 'demo.public.orders', 'properties.bootstrap.servers' = 'kafka:29092', 'properties.group.id' = 'flink-orders', 'scan.startup.mode' = 'earliest-offset', 'format' = 'json');


Créer la table source_order_items
CREATE TABLE source_order_items (after ROW<order_id STRING, product_id STRING, price DECIMAL(10, 2), freight_value DECIMAL(10, 2)>) WITH ('connector' = 'kafka', 'topic' = 'demo.public.order_items', 'properties.bootstrap.servers' = 'kafka:29092', 'properties.group.id' = 'flink-items', 'scan.startup.mode' = 'earliest-offset', 'format' = 'json');



Lancer la visualisation en temps réel
SELECT o.after.order_id, o.after.order_status, i.after.product_id, (i.after.price + i.after.freight_value) AS total_amount FROM source_orders o JOIN source_order_items i ON o.after.order_id = i.after.order_id;

SQL
CREATE TABLE sink_processed_sales (order_id STRING, order_status STRING, product_id STRING, total_amount DECIMAL(10, 2)) WITH ('connector' = 'kafka', 'topic' = 'demo.processed.sales', 'properties.bootstrap.servers' = 'kafka:29092', 'format' = 'json');

2. Lancer le Job en arrière-plan
INSERT INTO sink_processed_sales SELECT o.after.order_id, o.after.order_status, i.after.product_id, (i.after.price + i.after.freight_value) AS total_amount FROM source_orders o JOIN source_order_items i ON o.after.order_id = i.after.order_id;

Ligne A : Créer la table Engine Kafka (qui écoute le topic de Flink)

CREATE TABLE kafka_sales_queue (order_id String, order_status String, product_id String, total_amount Decimal(10,2)) ENGINE = Kafka SETTINGS kafka_broker_list = 'kafka:29092', kafka_topic_list = 'demo.processed.sales', kafka_group_name = 'clickhouse_flink_group', kafka_format = 'JSONEachRow';



Ligne B : Créer la table finale (qui stocke proprement pour Superset)

CREATE TABLE sales_analytics (order_id String, order_status String, product_id String, total_amount Decimal(10,2), processed_at DateTime DEFAULT now()) ENGINE = MergeTree() ORDER BY order_id;


Ligne C : Créer le pont (Materialized View) pour transférer les données
CREATE MATERIALIZED VIEW mv_kafka_to_analytics TO sales_analytics AS SELECT order_id, order_status, product_id, total_amount FROM kafka_sales_queue;
